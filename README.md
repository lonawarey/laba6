#include <stdio.h>

int main() {
    double A, result, term;
    int N, i;
    
    // Ввод данных без проверки корректности N
    printf("Введите вещественное число A: ");
    scanf("%lf", &A);
    
    printf("Введите целое число N (> 0): ");
    scanf("%d", &N);
    
    // Инициализация
    result = 1.0;      // Сумма начинается с 1
    term = 1.0;        // Первое слагаемое
    
    // Вычисление суммы с одним циклом, без условных операторов
    // Каждое следующее слагаемое = -предыдущее * A
    for (i = 1; i <= N; i++) {
        term = -term * A;  // Вычисляем новое слагаемое
        result = result + term;  // Добавляем к сумме
    }
    
    // Вывод результата
    printf("S = 1 - A + A^2 - A^3 + ... + (-1)^N * A^N = %.6f\n", result);
    
    return 0;
}
